Open FileName For Output As FileNumber
For i = 1 To LastRow
For j = 1 To LastCol



If j = LastCol Then
sLine = sLine & Cells(i, j).Value
Else
sLine = sLine & Cells(i, j).Value & Deliminator
End If

If i = 1 And j = 1 Then
sLine = ""
End If

If i = 2 And j = 2 Then
sLine = "|:---- |:----:| ----:"
End If

Next j

Print #FileNumber, sLine
sLine = ""

Next i
Close #FileNumber
MsgBox "Markdown table has been generated"
End Sub
